Generate PDF

Endpoint: /generate-pdf (POST)

Generates a PDF document from text content.

Request Body:

{
    "text_content": "This is the content of the PDF document."
}

Example Usage (JavaScript):


fetch('http://localhost:3009/generate-pdf', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
    },
    body: JSON.stringify({
        text_content: "Hello, PDFKit! This is a sample PDF generated via API."
    }),
})
.then(response => response.json())
.then(data => {
    console.log(data);
    if (data.filePath) {
        document.getElementById('pdfLink').innerHTML = `<a href="http://localhost:3009${data.filePath}" target="_blank">Download/View PDF</a>`;
    }
})
.catch(error => console.error('Error:', error));
